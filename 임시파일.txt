#include <stdio.h>
#include <ctype.h>

int main()
{
    int seen[26] = {0};
    int i;
    char ch;
    
    while (isalpha(ch = getchar())) {
        seen['a' - tolower(ch)] ++;
    }
    while (isalpha(ch = getchar())) {
        seen['a' - tolower(ch)] --;
    }
    for(i=0; i<26; i++){
        if(seen[i]!=0)
            printf;
    }
}
